\newcommand{\ChapterBannerNumbered}[1]{%
  \begin{tikzpicture}[remember picture,overlay]

    \path[
      shade,
      left color=chapterBlue,
      right color=chapterTeal,
      rounded corners=14pt
    ]
    (0,-4.2) rectangle (\textwidth,4.4);

    % CHAPTER label
    \node[
      anchor=west,
      text=chapterSky,
      font=\fontsize{12}{14}\selectfont\bfseries\sffamily
    ]
    at (1.0,2.7)
    {};

    % Huge chapter number
    \node[
      anchor=west,
      text=white,
      opacity=0.2,
      font=\fontsize{180}{184}\selectfont\bfseries\sffamily
    ]
    at (0.5,0.5)
    {\thechapter};

    % Chapter title
    % Shadow copy
    \node[
      anchor=west,
      align=left,
      text=black,
      opacity=0.55,
      font=\fontsize{26}{30}\selectfont\bfseries\sffamily,
      text width=0.55\textwidth
    ]
    at (5.26,0.04)
    {#1};

    % Main title
    \node[
      anchor=west,
      align=left,
      text=chapterSky,
      font=\fontsize{26}{30}\selectfont\bfseries\sffamily,
      text width=0.55\textwidth
    ]
    at (5.2,0.1)
    {#1};

    % Bottom label
    \path[
      fill=chapterSky,
      rounded corners=6pt
    ]
    (0.8,-3.55) rectangle (5.8,-2.9);

    \node[
      anchor=west,
      text=chapterBlue,
      font=\sffamily\bfseries\small
    ]
    at (1.0,-3.22)
    {PALPATINE LAB GUIDE};

  \end{tikzpicture}

  \vspace{4.5cm}
}

\newcommand{\SectionCard}[2]{%
  \tikz[baseline=(textnode.base)]{
    \node[
      fill=chapterSky,
      rounded corners=3pt,
      inner xsep=6pt,
      inner ysep=3pt
    ] (textnode) {\strut #1\quad #2};
  }%
}
